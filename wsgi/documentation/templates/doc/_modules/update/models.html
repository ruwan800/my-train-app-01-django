[docs]def addConflict(cat, value, msg):
    """ mention items with conflicts in conflict list. """
    trace = "{} - {} - {}".format(cat, value, msg)
    Q = Conflict.objects.filter(trace=trace)
    if Q:
        Q[0].resolved = False
        Q[0].save()
    else:
        Q1 = Conflict(cat=cat, value=value, msg=msg, trace=trace)
        Q1.save()
